68 struct switchtec_dev *switchtec_open(const char *device)
69 {
70  int idx;
71  int domain = 0;
72  int bus, dev, func;
73  char *endptr;
74  struct switchtec_dev *ret;
75 
76  if (strchr(device, '/') || strchr(device, '\\')) {
77  ret = switchtec_open_by_path(device);
78  goto found;
79  }
80 
81  if (sscanf(device, "%x:%x.%x", &bus, &dev, &func) == 3) {
82  ret = switchtec_open_by_pci_addr(domain, bus, dev, func);
83  goto found;
84  }
85 
86  if (sscanf(device, "%x:%x:%x.%x", &domain, &bus, &dev, &func) == 4) {
87  ret = switchtec_open_by_pci_addr(domain, bus, dev, func);
88  goto found;
89  }
90 
91  errno = 0;
92  idx = strtol(device, &endptr, 0);
93  if (!errno && endptr != device) {
94  ret = switchtec_open_by_index(idx);
95  goto found;
96  }
97 
98  if (sscanf(device, "switchtec%d", &idx) == 1) {
99  ret = switchtec_open_by_index(idx);
100  goto found;
101  }
102 
103  errno = ENODEV;
104  return NULL;
105 
106 found:
107  if (ret)
108  snprintf(ret->name, sizeof(ret->name), "%s", device);
109 
110  return ret;
111 }

266 int switchtec_status(struct switchtec_dev *dev,
267  struct switchtec_status **status)
268 {
269  uint64_t port_bitmap = 0;
270  int ret;
271  int i, p;
272  int nr_ports = 0;
273  struct switchtec_status *s;
274 
275  if (!status) {
276  errno = EINVAL;
277  return -errno;
278  }
279 
280  struct {
281  uint8_t phys_port_id;
282  uint8_t par_id;
283  uint8_t log_port_id;
284  uint8_t stk_id;
285  uint8_t cfg_lnk_width;
286  uint8_t neg_lnk_width;
287  uint8_t usp_flag;
288  uint8_t linkup_linkrate;
289  uint16_t LTSSM;
290  uint16_t reserved;
291  } ports[SWITCHTEC_MAX_PORTS];
292 
293  ret = switchtec_cmd(dev, MRPC_LNKSTAT, &port_bitmap, sizeof(port_bitmap),
294  ports, sizeof(ports));
295  if (ret)
296  return ret;
297 
298 
299  for (i = 0; i < SWITCHTEC_MAX_PORTS; i++) {
300  if ((ports[i].stk_id >> 4) > SWITCHTEC_MAX_STACKS)
301  continue;
302  nr_ports++;
303  }
304 
305  s = *status = calloc(nr_ports, sizeof(*s));
306  if (!s)
307  return -ENOMEM;
308 
309  for (i = 0, p = 0; i < SWITCHTEC_MAX_PORTS && p < nr_ports; i++) {
310  if ((ports[i].stk_id >> 4) > SWITCHTEC_MAX_STACKS)
311  continue;
312 
313  s[p].port.partition = ports[i].par_id;
314  s[p].port.stack = ports[i].stk_id >> 4;
315  s[p].port.upstream = ports[i].usp_flag;
316  s[p].port.stk_id = ports[i].stk_id & 0xF;
317  s[p].port.phys_id = ports[i].phys_port_id;
318  s[p].port.log_id = ports[i].log_port_id;
319 
320  s[p].cfg_lnk_width = ports[i].cfg_lnk_width;
321  s[p].neg_lnk_width = ports[i].neg_lnk_width;
322  s[p].link_up = ports[i].linkup_linkrate >> 7;
323  s[p].link_rate = ports[i].linkup_linkrate & 0x7F;
324  s[p].ltssm = le16toh(ports[i].LTSSM);
325  s[p].ltssm_str = ltssm_str(s[i].ltssm, 0);
326  p++;
327  }
328 
329  qsort(s, nr_ports, sizeof(*s), compare_status);
330 
331  return nr_ports;
332 }

438 static int log_a_to_file(struct switchtec_dev *dev, int sub_cmd_id, int fd)
439 {
440  int ret;
441  int read = 0;
442  struct log_a_retr_result res;
443  struct log_a_retr cmd = {
444  .sub_cmd_id = sub_cmd_id,
445  .start = -1,
446  };
447 
448  res.hdr.remain = 1;
449 
450  while (res.hdr.remain) {
451  ret = switchtec_cmd(dev, MRPC_FWLOGRD, &cmd, sizeof(cmd),
452  &res, sizeof(res));
453  if (ret)
454  return -1;
455 
456  ret = write(fd, res.data, sizeof(*res.data) * res.hdr.count);
457  if (ret < 0)
458  return ret;
459 
460  read += le32toh(res.hdr.count);
461  cmd.start = res.hdr.next_start;
462  }
463 
464  return read;
465 }

486  ret = write(fd, res.data, res.hdr.length);
487  if (ret < 0)
488  return ret;
489 
490  read += le32toh(res.hdr.length);
491  cmd.offset = htole32(read);
492  }
493 
494  return read;
