137 int switchtec_cmd(struct switchtec_dev *dev, uint32_t cmd,
138  const void *payload, size_t payload_len, void *resp,
139  size_t resp_len)
140 {
141  int ret;
142 
143  cmd &= SWITCHTEC_CMD_MASK;
144  cmd |= dev->pax_id << SWITCHTEC_PAX_ID_SHIFT;
145 
146  ret = dev->ops->cmd(dev, cmd, payload, payload_len, resp, resp_len);
147  if (ret > 0) {
148  mrpc_error_cmd = cmd & SWITCHTEC_CMD_MASK;
149  errno |= SWITCHTEC_ERRNO_MRPC_FLAG_BIT;
150  }
151 
152  return ret;
153 }
