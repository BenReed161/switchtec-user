164 int switchtec_fw_write_fd(struct switchtec_dev *dev, int img_fd,
165  int dont_activate,
166  void (*progress_callback)(int cur, int tot))
167 {
168  enum switchtec_fw_dlstatus status;
169  enum mrpc_bg_status bgstatus;
170  ssize_t image_size, offset = 0;
171  int ret;
172  struct cmd_fwdl cmd = {};
173 
174  image_size = lseek(img_fd, 0, SEEK_END);
175  if (image_size < 0)
176  return -errno;
177  lseek(img_fd, 0, SEEK_SET);
178 
179  switchtec_fw_dlstatus(dev, &status, &bgstatus);
180 
181  if (status == SWITCHTEC_DLSTAT_INPROGRESS) {
182  errno = EBUSY;
183  return -EBUSY;
184  }
185 
186  if (bgstatus == MRPC_BG_STAT_INPROGRESS) {
187  errno = EBUSY;
188  return -EBUSY;
189  }
190 
191  cmd.hdr.subcmd = MRPC_FWDNLD_DOWNLOAD;
192  cmd.hdr.dont_activate = !!dont_activate;
193  cmd.hdr.img_length = htole32(image_size);
194 
195  while (offset < image_size) {
196  ssize_t blklen = read(img_fd, &cmd.data,
197  sizeof(cmd.data));
198 
199  if (blklen == -EAGAIN || blklen == -EWOULDBLOCK)
200  continue;
201 
202  if (blklen < 0)
203  return -errno;
204 
205  if (blklen == 0)
206  break;
207 
208  cmd.hdr.offset = htole32(offset);
209  cmd.hdr.blk_length = htole32(blklen);
210 
211  ret = switchtec_cmd(dev, MRPC_FWDNLD, &cmd, sizeof(cmd),
212  NULL, 0);
213 
214  if (ret < 0)
215  return ret;
216 
217  ret = switchtec_fw_wait(dev, &status);
218  if (ret < 0)
219  return ret;
220 
221  offset += cmd.hdr.blk_length;
222 
223  if (progress_callback)
224  progress_callback(offset, image_size);
225 
226  }
227 
228  if (status == SWITCHTEC_DLSTAT_COMPLETES)
229  return 0;
230 
231  if (status == SWITCHTEC_DLSTAT_SUCCESS_FIRM_ACT)
232  return 0;
233 
234  if (status == SWITCHTEC_DLSTAT_SUCCESS_DATA_ACT)
235  return 0;
236 
237  if (status == 0)
238  return SWITCHTEC_DLSTAT_HARDWARE_ERR;
239 
240  return status;
241 }

688  unsigned long addr, size_t len,
689  void (*progress_callback)(int cur, int tot))
690 {
691  int ret;
692  unsigned char buf[(MRPC_MAX_DATA_LEN-8)*4];
693  size_t read = 0;
694  size_t total_len = len;
695  size_t total_wrote;
696  ssize_t wrote;
697 
698  while(len) {
699  size_t chunk_len = len;
700  if (chunk_len > sizeof(buf))
701  chunk_len = sizeof(buf);
702 
703  ret = switchtec_fw_read(dev, addr, chunk_len, buf);
704  if (ret < 0)
705  return ret;
706 
707  total_wrote = 0;
708  while (total_wrote < ret) {
709  wrote = write(fd, &buf[total_wrote],
710  ret - total_wrote);
711  if (wrote < 0)
712  return -1;
713  total_wrote += wrote;
714  }
715 
716  read += ret;
717  addr += ret;
718  len -= ret;
719 
720  if (progress_callback)
721  progress_callback(read, total_len);
722  }
723 
724  return read;
